#include <stdio.h>
int main() {
    int n;
    unsigned long long factorial = 1;
    printf("Enter a non-negative integer: ");
    scanf("%d", &n);
    if (n < 0) {
        printf("Error! Factorial of a negative number does not exist.\n");
    } else {
        int i = 1; 
        do {
            factorial *= i;
            i++;
        } while (i <= n);
        printf("Factorial of %d = %llu\n", n, factorial);
    }
    return 0;
}
