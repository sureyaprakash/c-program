#include <stdio.h>
#include <string.h>  
int main() {
    char str[200];
    int i = 0, j = 0;
    printf("Enter a string: ");
    fgets(str, sizeof(str), stdin);
    str[strcspn(str, "\n")] = '\0';
    while (str[i]) {
        if (str[i] != ' ') {
            str[j++] = str[i];
        }
        i++;
    }
    str[j] = '\0'; 
    printf("String without spaces: %s\n", str);
    return 0;
}
