#include <stdio.h>
int main() {
    int mat1[3][3], mat2[3][3], matSub[3][3];
    int i, j;
    printf("Enter elements of first 3x3 matrix:\n");
    for(i = 0; i < 3; i++) {
        for(j = 0; j < 3; j++) {
            scanf("%d", &mat1[i][j]);
        }
    }
    printf("Enter elements of second 3x3 matrix:\n");
    for(i = 0; i < 3; i++) {
        for(j = 0; j < 3; j++) {
            scanf("%d", &mat2[i][j]);
        }
    }
    for(i = 0; i < 3; i++) {
        for(j = 0; j < 3; j++) {
            matSub[i][j] = mat1[i][j] - mat2[i][j];
        }
    }
    printf("Subtraction of the matrices:\n");
    for(i = 0; i < 3; i++) {
        for(j = 0; j < 3; j++) {
            printf("%d ", matSub[i][j]);
        }
        printf("\n");
    }
    return 0;
}
