#include <stdio.h>
#include <ctype.h>
#include <string.h>
int main() {
    char str[200];
    int i, wordCount = 0;
    int inWord = 0;
    printf("Enter a string: ");
    fgets(str, sizeof(str), stdin);
    str[strcspn(str, "\n")] = '\0';

    for (i = 0; str[i] != '\0'; i++) {
        if (isspace(str[i])) {
            inWord = 0;
        } else if (inWord == 0) {
            inWord = 1;
            wordCount++;
        }
    }
    printf("Word count: %d\n", wordCount);

    return 0;
}
