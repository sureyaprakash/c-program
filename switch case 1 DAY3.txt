#include <stdio.h>
int main() {
    int choice;
    printf("Enter a number (1-3): ");
    scanf("%d", &choice);
    switch (choice) {
        case 1:
            printf("You choose Option 1.\n");
            break;
        case 2:
            printf("You choose Option 2.\n");
            break;
        case 3:
            printf("You choose Option 3.\n");
            break;
        default:
            printf("Invalid choice. Please enter a number between 1 and 3.\n");
            break;
    }
    return 0;
}
